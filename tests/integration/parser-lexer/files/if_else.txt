## Sentencias If-Else:
    Se muestra una sentencia if e if-else válida
##

PROGRAM
{
    uint X;

    X := 1UI;

    do {
        if (X < 5UI)
            print("Menor a 5");
        else if (X == 5UI)
            print("Igual a 5");
        else
            print("Mayor a 5");
        endif; endif;
    } while (X <= 10UI);

    if (X > 10UI) {
        print("No es posible");
    } endif;
}
