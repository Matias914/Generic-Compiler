// Regla inicial
program:
    IDENTIFIER '{' body '}'
;

body:
    /*    vacio    */
    | executable_block body
    | declarative_block body
;

control_block:
    DO executable_block WHILE '(' condition ')';
;

condition:
    expression comparator expression
;

expression:
    IDENTIFIER

comparator:
    GE_OP
    | LE_OP
    | EQUAL_OP
    | NOT_EQUAL_OP
;

/*    declarative    */

declarative_block:
    declarative_sentence ';'
    | declarative_block declarative_sentence ';'
;

declarative_sentence:
    type IDENTIFIER
    | type IDENTIFIER variable_list
;

variable_list:
    ',' IDENTIFIER
    | variable_list ',' IDENTIFIER
;

*    executable    */

executable_block:
    executable_sentence ';'
    | executable_sentence executable_sentence ';'
;

executable_sentence:
    IDENTIFIER ASSIGN_OP arithmetic_expression
;

arithmetic_expression:
    IDENTIFIER '+' IDENTIFIER
;

identifier:
    IDENTIFIER
;
